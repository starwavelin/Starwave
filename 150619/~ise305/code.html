The Statistics of Charges on Patients


                    SELECT Patient.PatSSN, Patient.PatFirstName,
                         Patient.PatLastName, OperationRoom.OptRoomNo,
                            Building.BldgName, Operation.OperationFee
                    FROM Patient INNER JOIN
                        ((Building INNER JOIN OperationRoom ON Building.[BldgNo]
                         = OperationRoom.[BldgNo]) INNER JOIN Operation ON
                        OperationRoom.[OptRoom_key] = Operation.[OptRoom_key])
                         ON Patient.[PatSSN] = Operation.[PatSSN]
                    WHERE (((Building.BldgName)="General Surgery Center")
                        AND ((Operation.OperationFee)>1000));

Physicians who Performed "Brain-related" Operations


                    SELECT DISTINCT Physician.PhyNo, Employee.EmpFirstName,
                            Employee.EmpLastName, Operation.OptDate,
                            Operation.OptTime, Operation.OptName,
                            Patient.PatFirstName, Patient.PatLastName
                    FROM Patient INNER JOIN ((Employee INNER JOIN Physician ON
                        Employee.[EmpSSN] = Physician.[EmpSSN]) INNER JOIN
                        Operation ON Physician.[PhyNo] = Operation.[PhyNo]) ON
                        Patient.[PatSSN] = Operation.[PatSSN]
                    WHERE (((Operation.OptName) Like '*Brain*'))
                    GROUP BY Physician.PhyNo, Employee.EmpFirstName,
                            Employee.EmpLastName, Operation.OptDate,
                            Operation.OptTime, Operation.OptName,
                            Patient.PatFirstName, Patient.PatLastName;
