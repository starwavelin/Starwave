Basic entity relations:

  • person 1--has--1 account
  • person 1--takes--1 major
  • major 1--has--* major_requirement 1--has--* course_set
  • major 1--has--* course 1--has--* course_substitutes

Example Code of Tables

/* after login, create a new database called "rumast" */


                    create database rumast;

/* select the database */


                    use rumast;

/* create tables */

/* it stores major title, ex: CSE, ISE, etc. */


                    create table major(
                        title char(5) not null,
                        primary key (title)
                    )type=innodb;

/* Person data, which can be used for administrator, secretary and students. For administrator and secretary, we only need their first names and last names; for students, all fields are required. */


                    create table Person(
                        id      integer     auto_increment,
                        firstName   char(25)    not null,
                        lastName    char(25)    not null,
                        major       char(5),
                        matriculateYear     integer,
                        matriculateSemester char(2),
                        versionCode         char(20),
                        primary key         (id),
                        foreign key (major) references major (title)
                        on delete no action
                        on update cascade
                    )type=innodb;
